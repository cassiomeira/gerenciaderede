# Cria o Script Inteligente
/system script add name="BackupAutoFTP" policy=ftp,reboot,read,write,policy,test,password,sniff,sensitive source="# --- 1. Lista de interfaces por ordem de prioridade ---
:local interfaces {\"bdg_wan\"; \"ether1\"; \"wlan1\"; \"lo\"; \"sfp1\"; \"sfpplus1\"; \"bdg_lan\"}
:local ipComMascara \"0.0.0.0/0\"
:local achouIP false

# --- 2. Busca automatica e inteligente nas interfaces ---
:foreach int in=\$interfaces do={
    :if (!\$achouIP) do={
        :local idBusca [/ip address find interface=\$int]
        :if ([:len \$idBusca] > 0) do={
            :set ipComMascara [/ip address get [:pick \$idBusca 0] address]
            :set achouIP true
        }
    }
}

# --- 3. Remove a mascara e monta o nome do arquivo ---
:local ipSemMascara [:pick \$ipComMascara 0 [:find \$ipComMascara \"/\"]]
:local nomeBase ([/system identity get name] . \"_\" . \$ipSemMascara)

# --- 4. Backup em formato texto (.rsc) ---
:global namefilersc (\$nomeBase . \".rsc\")
/export file=\$namefilersc
:delay 10s
/tool fetch address=217.216.86.231 user=matrix password=suporte10025 src-path=\"\$namefilersc\" mode=ftp upload=yes port=21 dst-path=\"Backup_Mikrotik/\$namefilersc\"
:delay 50s
/file remove \$namefilersc

# --- 5. Backup binario (.backup) ---
:global namefilebkp (\$nomeBase . \".backup\")
/system backup save name=\$namefilebkp password=10025-MATRIX
:delay 10s
/tool fetch address=192.168.13.2 user=Netcar password=1957ntc src-path=\"\$namefilebkp\" mode=ftp upload=yes port=21 dst-path=\"Backup_Mikrotik/\$namefilebkp\"
:delay 50s
/file remove \$namefilebkp"

# Cria o Agendamento
/system scheduler add name="Rotina_Backup_3AM" start-time=03:00:00 interval=1d on-event="BackupAutoFTP" policy=ftp,reboot,read,write,policy,test,password,sniff,sensitive

:put "Script e Agendamento instalados com sucesso!"
